FILE=/usr/bin/gconftool-2

[ -f $FILE ] || exit 0

/usr/bin/gconftool-2 \
	 --direct \
	 --config-source xml:readwrite:/etc/gconf/gconf.xml.mandatory \
   --type int \
   --set /apps/gnome-screensaver/idle_delay 15 && exit 1
